// server.js
const express = require('express');
const cors = require('cors');
require('dotenv').config();

const app = express();
app.use(cors());
app.use(express.json());

app.post('/api/generate-topics', async (req, res) => {
  try {
    const { domain, department, researchType, studyArea, keywords } = req.body;
    
    const prompt = `Generate 6 innovative research proposal topics for:

Domain: ${domain}
Department: ${department}
Research Type: ${researchType || 'Any appropriate methodology'}
Study Area: ${studyArea}
Keywords: ${keywords || 'None specified'}

Requirements:
- Each topic should be specific, measurable, and achievable
- Address current gaps in healthcare/public health
- Consider ethical feasibility and resource constraints
- Mix different research methodologies
- Relevant to current medical challenges
- 50-80 words per topic

Format: Return exactly 6 numbered research titles (1. 2. 3. etc.) without additional commentary.`;

    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent?key=${process.env.GEMINI_API_KEY}`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        contents: [{
          parts: [{
            text: prompt
          }]
        }],
        generationConfig: {
          temperature: 0.8,
          topK: 40,
          topP: 0.95,
          maxOutputTokens: 1024
        }
      })
    });

    const data = await response.json();
    
    if (data.candidates && data.candidates[0]) {
      const generatedText = data.candidates[0].content.parts[0].text;
      const topics = parseTopics(generatedText);
      
      res.json({ success: true, topics });
    } else {
      throw new Error('No topics generated');
    }
    
  } catch (error) {
    console.error('Gemini API Error:', error);
    res.status(500).json({ success: false, error: 'Failed to generate topics' });
  }
});

function parseTopics(text) {
  const lines = text.split('\n').filter(line => line.trim());
  const topics = [];
  
  lines.forEach(line => {
    const match = line.match(/^\d+\.\s*(.+)/);
    if (match) {
      topics.push(match[1].trim());
    }
  });
  
  return topics.slice(0, 6);
}

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
